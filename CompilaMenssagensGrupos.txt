from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
import time

# Configura o driver
service = Service(ChromeDriverManager().install())
driver = webdriver.Chrome(service=service)

# Abre o WhatsApp Web
driver.get('https://web.whatsapp.com')
time.sleep(15)  # Tempo para escanear o QR code

# Seleciona o grupo pelo nome (substitua '1°/98 VI COMAR ALOJ. B')
group_title = '1°/98 VI COMAR ALOJ. B'
group = driver.find_element(By.XPATH, f"//span[@title='{group_title}']")
group.click()

# Coleta as últimas mensagens
messages = driver.find_elements(By.CSS_SELECTOR, "div.message-in")
for message in messages:
    print(message.text)

driver.quit()
